JavaScript Array Metodları:
push() & pop() & unshift() & shift()

push() Metodu

push: dizinin sonuna eleman ekler. Birden fazla eleman eklenebilir. Dizinin eleman eklendikten sonraki eleman sayısını döndürür.

let dizi1 = ["a","b","c"]

dizi1.push("ç") // 4

dizi1 // ["a","b","c","ç"]

dizi1.push("d","e") // 6

dizi1 // ["a","b","c","ç","d","e"]

pop() Metodu

pop: dizinin son elemanını siler. Dizinin silinen elemanını döndürür. Dizinin uzunluğu 1 azalmış olur.

let dizi2 = ["f","g","ğ"]

dizi2.pop() // "ğ"

dizi2 // ["f","g"]

unshift() Metodu

unshift: dizinin başına eleman ekler. Birden fazla eleman eklenebilir. Dizinin eleman eklendikten sonraki eleman sayısını döndürür.

let dizi3 = ["h","ı","i"]

dizi3.unshift("j") // 4

dizi3 // ["h","ı","i","j"]

dizi3.unshift("k","l")

dizi3 // ["h","ı","i","j","k","l"]

shift() Metodu

pop: dizinin ilk elemanını siler. Dizinin silinen elemanını döndürür. Dizinin uzunluğu 1 azalmış olur.

let dizi4 = ["m","n","o"]

dizi4.shift() // 2

dizi4 // ["n","o"]
